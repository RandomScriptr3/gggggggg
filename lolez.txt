local hook
hook = hookmetamethod(game, "__namecall", function(...)
    local args = {...}
    local self = args[1]
    if tostring(self) == "AdminGUI" or tostring(self) == "WalkSpeedChanged" or tostring(self) == "Ban" then
        return nil
    end
    if settings["inf golden"] == true then
        if tostring(self) == "GoldenHit" then
            args[3] = true
            return hook(unpack(args))
        end
        if tostring(self) == "Goldify" then
            if args[3] then
                return nil
            end
        end
    end
    if settings["zahand kick"] == true then
        if tostring(self) == "Erase" then
            coroutine.wrap(function()
                pcall(function()
                    task.wait(0.45)
                    lp.Character.HumanoidRootPart.CFrame = CFrame.new(brazil.portal.Position + Vector3.new(0, 5, 0))
                end)
            end)()
        end
    end
    return hook(...)
end)

local currentanim

lp.Chatted:Connect(function(msg)
    pcall(function()
        local check = string.split(string.lower(msg), "/e ")[2]
        if check and settings["animations"] == true then
            local anim = anims[string.upper(check)]
            if anim then
                game.ReplicatedStorage.AnimationSound:FireServer(table.find(anims, string.upper(check)))
                if currentanim then
                    currentanim:Stop()
                end
                local a = lp.Character.Humanoid:LoadAnimation(anim)
                a.Priority = Enum.AnimationPriority.Idle
                a:Play()
                currentanim = a
            end
        end
    end)
end)

    local hm = lp.Character:FindFirstChildOfClass("Humanoid")
        if hm.MoveDirection ~= Vector3.new(0, 0, 0) then
            if currentanim then
                currentanim:Stop()
            end
        end
        if settings["slap aura"] == true then
            local closest = 25
            local lstedplrs = {}
            for _, player in pairs(game.Players:GetChildren()) do
                if player ~= lp then
                    local rock = player.Character:FindFirstChild("rock")
                    local uno = player.Character.Head:FindFirstChild("UnoReverseCard")
                    local dist = lp:DistanceFromCharacter(player.Character.PrimaryPart.Position)
                    if dist < closest and not rock and not uno then
                        table.insert(lstedplrs, player)
                    end
                end
            end
            local glov = lp.Character:FindFirstChildOfClass("Tool") or lp.Backpack:FindFirstChildOfClass("Tool")
            if glov then
                if not table.find(sgn, glov.Name) then
                    local d = decompile(glov.LocalScript)
                    local sgtl = false
                    local repline = ""
                    for _, line in pairs(string.split(d, "\n")) do
                        if string.split(line, 'script.Parent:WaitForChild("Glove").Touched')[2] then
                            sgtl = true
                        end
                        if string.split(line, "ReplicatedStorage")[2] and sgtl then
                            repline = line
                            break
                        end
                    end
                    local remotename = string.split(string.split(repline, ".")[2], ":")[1]
                    local remote = game.ReplicatedStorage[remotename]
                    table.insert(sgn, glov.Name)
                    table.insert(sgr, remote)
                elseif #lstedplrs > 0 then
                    for _, player in pairs(lstedplrs) do
                        local remote = sgr[table.find(sgn, glov.Name)]
                        remote:FireServer(player.Character.Head)
                    end

            end
